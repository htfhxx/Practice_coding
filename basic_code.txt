struct Date{
    int year,month,day;
    void nextDay(){
        day++;
        int yeap=0;
        if(year%400==0 || year%100!=0 &&year%4==0) yeap=1;
        if(days[month][yeap]<day){
            day=1;
            month++;
        }
        if(month>12){
            month=1;
            year++;
        }
    }
};
int buf[5005][13][32];

	scanf("%lf",&num);
	if(c=='+')a[cnt++]=num;
	else if(c=='-')a[cnt++]=-num;
	else if(c=='*')a[cnt-1]*=num;
	else if(c=='/')a[cnt-1]/=num;
}
